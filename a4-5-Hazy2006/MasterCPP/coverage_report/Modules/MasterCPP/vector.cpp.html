/// <summary>
/// Doubles the internal capacity of the dynamic vector and copies existing elements to the newly allocated memory.
/// </summary>
void DynamicVector::_resize()
{
	this->capacity *= 2;
	TElem* newBlock = new TElem[this->capacity];

}

/// <summary>
/// Adds an element to the end of the dynamic vector.
/// </summary>
/// <param name="element">The element to add to the vector.</param>
void DynamicVector::add(TElem element)
{
	if (this->size == this->capacity) {
		this->_resize();
	}
	this->elems[this->size] = element;
	this->size++;
}


/// <summary>
/// Removes the element at the specified index from the dynamic vector.
/// </summary>
/// <param name="index">The zero-based index of the element to remove. If the index is out of bounds, no action is taken.</param>
void DynamicVector::remove(int index)
{
	if (index < 0 || index >= this->size) {
		return; 
	}

	for (int i = index; i < this->size - 1; ++i) {
		this->elems[i] = this->elems[i + 1];
	}
	this->size--;
}
